function rgb(r, g, b) {
	function rgbTohex(n) {

		let x = Math.round(Number(n));

		if (x < 0) {
			x = 0;
		} else if (x > 255) {
			x = 255;
		}
	
		let hex = x.toString(16);
		if (hex.length < 2) {
			hex = "0" + hex;
		}
		return hex;
	}

	let red = rgbTohex(r).toUpperCase();
	let green = rgbTohex(g).toUpperCase();
	let blue = rgbTohex(b).toUpperCase();

	return red + green + blue;
}
					
				
					
function towelSort(matrix) {

	let result = [];
	let i = 0;

	if(arguments[0] === undefined ){
		return result
	}else{
		while (i < matrix.length) {
		if (i % 2 === 0) {
			result = result
			.concat(matrix[i]);
			
		} else {
			result = result
			.concat(matrix[i]
			.reverse());
		}
		i++;
	}
	}

	return result;
};
